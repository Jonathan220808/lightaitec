        let nowRoundIndex = 0;
        let selectArr = [];
        let allPlayerSelect = [];
        let currentPlayer = 1;
        const totalPlayer = 4;

        function initGame(){
            selectArr = [];
            currentPlayer = 1;
            allPlayerSelect = [];
            document.getElementById("result").innerText = "";
            document.getElementById("nextBtn").style.display = "none";
            document.getElementById("sureBtn").disabled = true;
            document.getElementById("tip").innerText = `玩家${currentPlayer}：请选择 2 张卡牌`;
            renderCard();
        }

        function renderCard(){
            clearCardActive();
            selectArr = [];
            let html = "";
            let cards = cardList[nowRoundIndex];
            for(let i=0;i<5;i++){
                html += `
                <div class="card" onclick="chooseCard(${i},this)">
                    <p>${cards[i]}</p >
                </div>
                `;
            }
            document.getElementById("cardBox").innerHTML = html;
            document.getElementById("sureBtn").disabled = true;
        }

        function chooseCard(index,dom){
            if(selectArr.length >= 2 && !dom.classList.contains("active")) return;
            if(dom.classList.contains("active")){
                dom.classList.remove("active");
                let idx = selectArr.indexOf(index);
                selectArr.splice(idx,1);
            }else{
                dom.classList.add("active");
                selectArr.push(index);
            }
            document.getElementById("sureBtn").disabled = selectArr.length !== 2;
        }

        function sureSelect(){
            allPlayerSelect.push([...selectArr]);
            if(currentPlayer < totalPlayer){
                currentPlayer++;
                clearCardActive();
                selectArr = [];
                document.getElementById("tip").innerText = `玩家${currentPlayer}：请选择 2 张卡牌`;
                document.getElementById("sureBtn").disabled = true;
            }else{
                judgeResult();
                document.getElementById("nextBtn").style.display = "inline-block";
                document.getElementById("sureBtn").disabled = true;
            }
        }

        function judgeResult(){
            let standard = allPlayerSelect[0].sort((a,b)=>a-b);
            let resText = "本轮结果公示：<br/>";
            for(let i=1;i<totalPlayer;i++){
                let playerSelect = allPlayerSelect[i].sort((a,b)=>a-b);
                if(playerSelect[0]===standard[0] && playerSelect[1]===standard[1]){
                    resText += `✅ 玩家${i+1}：完全答对！与玩家1选择一致<br/>`;
                }else{
                    resText += `❌ 玩家${i+1}：选择不一致，回答错误<br/>`;
                }
            }
            document.getElementById("result").innerHTML = resText;
        }

        function clearCardActive(){
            let allCard = document.querySelectorAll(".card");
            allCard.forEach(item=>{
                item.classList.remove("active");
            });
        }

        function nextRound(){
            nowRoundIndex++;
            if(nowRoundIndex >= cardList.length) nowRoundIndex = 0;
            initGame();
        }

        function fullReset(){
            let randomIndex = Math.floor(Math.random() * cardList.length);
            while(randomIndex === nowRoundIndex){
                randomIndex = Math.floor(Math.random() * cardList.length);
            }
            nowRoundIndex = randomIndex;
            initGame();
        }

        window.onload = initGame;
